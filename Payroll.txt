package Company;

class Payroll {
	// Use only the given instance variables
	public String month = "";
	int exp_work = 0; // expect work day
	int act_work = 0; // actual work
	int day_absent = 0;
	double hour_per_day = 0;
	private double hour_rate = 0;
	double total_pay = 0;

	public double cal_Pay(double hour, 
			      double hour_rate){
	     return hour * hour_rate;
	}
	public int get_Absent(int expect_day, 
				int actual_day){
	     if ( (expect_day - actual_day) < 0 ){
		return 0;
	     }
	     else {
		return expect_day - actual_day;
	     }
	}
	public void set_hour_rate(double rate) {
		hour_rate = rate;
	}
	public double get_hour_rate() {
		return hour_rate;
	}
} // Payroll
